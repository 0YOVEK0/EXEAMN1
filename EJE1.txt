Hacer un programa que pida 3 numeros(7,1,3) y te diga cual es el mayor y cual es el menor

#include <iostream>

using namespace std;
int
main()
{
    int num1, num2, num3;
    cout << "Ingrese el primer numero: ";
    cin >> num1;

    cout << "Ingrese el segundo numero: ";
    cin >> num2;

    cout << "Ingrese el tercer numero: ";
    cin >> num3;



    int mayor = num1;

    int menor = num1;

    if (num2 > mayor)
    {
        mayor = num2;
    }
    if (num3 > mayor)
    {
        mayor = num3;
    }
    if (num2 < menor)
    {
        menor = num2;
    }
    if (num3 < menor)
    {
        menor = num3;
    }

    cout << "El numeroo mayor es: " << mayor << endl;
    cout << "El numero menor es: " << menor << endl;

}
